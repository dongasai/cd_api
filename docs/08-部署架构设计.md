      context: .

      - .:/var/www/cdapi
    depends_on:

      - redis
    networks:
      - cdapi-network

    ports:

      - "443:443"

      - app
    networks:

    networks:
      - cdapi-network

    volumes:
      - redis-data:/data

      - cdapi-network

  cdapi-network:

## 三、Nginx 配置

### 3.1 基础配置

```
┌─────────────────────────────────────────────────────────────────────┐
│                        Nginx 配置                                    │
└─────────────────────────────────────────────────────────────────────┘

upstream cdapi {
    least_conn;
    server 10.0.0.1:9000 weight=5;
    server 10.0.0.2:9000 weight=5;
    server 10.0.0.3:9000 weight=5;
    keepalive 32;
}

server {
    listen 80;
    listen 443 ssl http2;
    server_name api.example.com;

    # SSL 配置
    ssl_certificate /etc/ssl/certs/api.example.com.crt;
    ssl_certificate_key /etc/ssl/private/api.example.com.key;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 1d;

    # 安全 Headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # 日志
    access_log /var/log/nginx/cdapi.access.log json;
    error_log /var/log/nginx/cdapi.error.log;

    # 根目录
    root /var/www/cdapi/public;
    index index.php;

    # Gzip 压缩
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_proxied any;
    gzip_types application/json text/plain;

    # API 端点
    location /v1/ {
        try_files $uri $uri/ /index.php?$query_string;
    }

    # PHP 处理
    location ~ \.php$ {
        fastcgi_pass cdapi;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include fastcgi_params;
        
        # 超时配置
        fastcgi_read_timeout 120s;
        fastcgi_send_timeout 120s;
        
        # 缓冲配置 (SSE 需要禁用)
        fastcgi_buffering off;
        fastcgi_cache off;
    }

    # SSE 流式响应
    location ~ ^/v1/(chat/completions|messages)$ {
        if ($request_method = POST) {
            fastcgi_buffering off;
            fastcgi_cache off;
            proxy_buffering off;
        }
        try_files $uri $uri/ /index.php?$query_string;
    }

    # 健康检查
    location /health {
        access_log off;
        return 200 "OK";
        add_header Content-Type text/plain;
    }

    # 禁止访问隐藏文件
    location ~ /\. {
        deny all;
    }
}
```

## 八、CI/CD 流水线

```
┌─────────────────────────────────────────────────────────────────────┐
│                        CI/CD 流水线                                  │
└─────────────────────────────────────────────────────────────────────┘

GitHub Actions:
name: Deploy

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3
    
    - name: Setup PHP
      uses: shivammathur/setup-php@v2
      with:
        php-version: '8.2'
        extensions: mbstring, xml, mysql, redis
    
    - name: Install Dependencies
      run: composer install --no-interaction
    
    - name: Run Tests
      run: php artisan test --parallel

  build:
    needs: test
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3
    
    - name: Build Docker Image
      run: |
        docker build -t cdapi:${{ github.sha }} .
        docker tag cdapi:${{ github.sha }} cdapi:latest
    
    - name: Push to Registry
      run: |
        echo ${{ secrets.DOCKER_PASSWORD }} | docker login -u ${{ secrets.DOCKER_USER }} --password-stdin
        docker push cdapi:${{ github.sha }}
        docker push cdapi:latest

  deploy:
    needs: build
    runs-on: ubuntu-latest
    steps:
    - name: Deploy to Production
      run: |
        kubectl set image deployment/cdapi cdapi=cdapi:${{ github.sha }}
        kubectl rollout status deployment/cdapi
```

数据库备份:
├─── 全量备份: 每日凌晨 2:00
├─── 增量备份: 每小时
├─── 保留策略: 7 天全量 + 30 天增量
└─── 异地备份: 同步到对象存储

备份脚本:
#!/bin/bash
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backup/mysql"
MYSQL_USER="backup"
MYSQL_PASS="password"

# 全量备份
mysqldump -u$MYSQL_USER -p$MYSQL_PASS --single-transaction --routines --triggers cdapi | gzip > $BACKUP_DIR/cdapi_$DATE.sql.gz

# 上传到对象存储
aws s3 cp $BACKUP_DIR/cdapi_$DATE.sql.gz s3://backup-bucket/mysql/

# 清理旧备份
find $BACKUP_DIR -name "*.sql.gz" -mtime +7 -delete
